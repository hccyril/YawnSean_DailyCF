**提示 1：** 考虑每个质因子，我们有哪些操作？

**提示 2：** 实际上需要考虑哪些质因子。

每个质因子的操作可以视为相互独立的，对最后结果产生贡献。

对于某个质数，我们考虑每个元素的次数。我们可以枚举最终 $gcd$ 中该质因子的次数，计算这个质因子要达到次数 $\alpha$ 的情况下的最小操作次数。我们预处理这样的 $alpha$ 及其次数的元素对。

我们需要处理多少这样的对呢？

因为如果质数 $p$ 有用，则其总出现次数一定是不小于 $n$ 的，否则没办法让每个数都是它的倍数。

进一步，如果 $p$ 的总次数和为 $S$ ，则我们最终需要考虑的它的次数是不超过 $\lfloor S/p\rfloor$ 的正整数。

需要考虑的质因子次数的和乘以 $n$ 不超过原本的各个质数的次数和，而后者不超过每个元素的次数和，也就是 $\mathcal{O}(n\log M)$ 的，所以我们需要考虑的转移只有 $\mathcal{O}(\log M)$ 个。

最后在 $\mathcal{O}(k)$ 个状态中，进行类似背包 DP 的转移即可，DP 的变量也就是操作不超过 $v$ 次能得到的最大数值。

按照不同的质因子分别进行转移，每次枚举当前质因子的操作次数 $\alpha$ ，用 $dp[i]\times p^\alpha$ 更新 $dp[i+operation_count]$ 。

时间复杂度为 $\mathcal{O}(M+(n+k)\log M)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    M = 10 ** 6 + 5
    pr = list(range(M))
    
    for i in range(2, M):
        if pr[i] == i:
            for j in range(i, M, i):
                pr[j] = i
    
    n, k = MII()
    nums = LII()
    
    cnt = [0] * M
    
    for x in nums:
        while x > 1:
            cnt[pr[x]] += 1
            x //= pr[x]
    
    dp = [1] * (k + 1)
    
    for i in range(M):
        if cnt[i] >= n:
            
            pws = [0] * 20
            for x in nums:
                pw = 0
                while x % i == 0:
                    x //= i
                    pw += 1
                pws[pw] += 1
    
            transitions = []
            val = 1
            
            for a in range(cnt[i] // n + 1):
                cost = 0
                for b in range(a):
                    cost += (a - b) * pws[b]
                transitions.append((cost, val))
                val *= i
            
            for i in range(k, -1, -1):
                dp_val = dp[i]
                for cost, val in transitions:
                    if i >= cost:
                        dp_val = fmax(dp_val, dp[i - cost] * val)
                dp[i] = dp_val
    
    print(' '.join(map(str, dp[1:])))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int M = 1e6 + 5;
	vector<int> pr(M);
	iota(pr.begin(), pr.end(), 0);

	for (int i = 2; i < M; i ++) {
		if (pr[i] == i) {
			for (int j = i; j < M; j += i) {
				pr[j] = i;
			}
		}
	}

	int n, k;
	cin >> n >> k;

	vector<int> nums(n);
	for (auto &x: nums) cin >> x;

	vector<int> cnt(M, 0);

	for (auto x: nums) {
		while (x > 1) {
			cnt[pr[x]] ++;
			x /= pr[x];
		}
	}

	vector<int> dp(k + 1, 1);

	for (int i = 0; i < M; i ++) {
		if (cnt[i] >= n) {
			vector<int> pws(20, 0);

			for (auto &x: nums) {
				int pw = 0;
				while (x % i == 0) {
					x /= i;
					pw ++;
				}
				pws[pw] ++;
			}

			vector<pair<int, int>> transitions;
			int val = 1;

			for (int a = 1; a <= cnt[i] / n; a ++) {
				val *= i;
				int cost = 0;
				for (int b = 0; b < a; b ++)
					cost += (a - b) * pws[b];
				transitions.emplace_back(cost, val);
			}

			for (int i = k; i >= 0; i --) {
				int dp_val = dp[i];
				for (auto &[cost, val]: transitions) {
					if (i >= cost) {
						dp_val = max(dp_val, dp[i - cost] * val);
					}
				}
				dp[i] = dp_val;
			}
		}
	}

	for (int i = 1; i <= k; i ++) cout << dp[i] << ' ';

	return 0;
}
```
