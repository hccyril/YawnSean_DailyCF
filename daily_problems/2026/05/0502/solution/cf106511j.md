**提示 1：** 每个人有何决策？

**提示 2：** 博弈如何考虑？

每个人的走的循环节就是对应的环的长度，最后的结果就是这个长度的最小公倍数。

所以每个人的实质选择是环的长度。

而环的总长度为 $n$ ，所以最多有 $\mathcal{O}(\sqrt{n})$ 种不同的环长，我们可以预处理它们。接下来我们可以枚举两个人所有可能选择的环长对。

如果第一个人的选择确定了，第二个人会选择能最小化结果的决策。因此对于每一种第一个人的决策，第二个人可以给出一个答案。

而第一个人只需在这些答案中取最大值即可。

也是博弈问题从后往前考虑的一个很好的例子——从最后一个决策的人出发往前推，因为最后一个人很容易决策。

时间复杂度为 $\mathcal{O}(n\log n)$ 。

### 具体代码如下——

Python 做法如下——

```Python []
def main():
    n = II()
    v1 = LGMI()
    v2 = LGMI()
    
    def find_cycles(x):
        vis = [0] * n
        possible = [0] * (n + 1)
        
        for i in range(n):
            if not vis[i]:
                cur = i
                cnt = 0
    
                while not vis[cur]:
                    vis[cur] = 1
                    cnt += 1
                    cur = x[cur]
                
                possible[cnt] = 1
        
        return [x for x in range(n + 1) if possible[x]]
    
    w1 = find_cycles(v1)
    w2 = find_cycles(v2)
    
    print(max(min(math.lcm(a, b) for b in w2) for a in w1))
```

C++ 做法如下——

```cpp []
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);

	int n;
	cin >> n;

	vector<int> v1(n), v2(n);
	for (auto &x: v1) cin >> x, x --;
	for (auto &x: v2) cin >> x, x --;

	auto find_cycles = [&] (vector<int> &perm) -> vector<int> {
		vector<int> vis(n, 0), possible(n + 1, 0);

		for (int i = 0; i < n; i ++) {
			if (!vis[i]) {
				int cur = i, cnt = 0;
				while (!vis[cur]) {
					vis[cur] = 1;
					cnt ++;
					cur = perm[cur];
				}
				possible[cnt] = 1;
			}
		}

		vector<int> ans;
		for (int i = 1; i <= n; i ++) {
			if (possible[i]) {
				ans.emplace_back(i);
			}
		}

		return ans;
	};

	auto w1 = find_cycles(v1);
	auto w2 = find_cycles(v2);

	long long ans = 0;

	for (auto &x: w1) {
		long long res = 1e18;
		for (auto &y: w2) {
			res = min(res, 1ll * x * y / gcd(x, y));
		}
		ans = max(ans, res);
	}

	cout << ans;

	return 0;
}
```
